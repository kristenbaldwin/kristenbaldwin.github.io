Loops Exercises

# Exercise 1

Using a for loop and the range function, print out the numbers between 1 and 10 inclusive, one on a line.


    for number in range(1, 11):
        print(number)

# Exercise 2

Same as the previous problem, except you will prompt the user for the number to start on and the number to end on.


    start = int(input("Start from: "))
    end = int(input("End on: "))
    for number in range(start, end + 1):
        print(number)

# Exercise 3

Print each odd number between 1 and 10 inclusive.


    for number in range(1, 11, 2):
        print(number)
            
    # OR 
            
    for number in range(0, 10):
        if number % 2 != 0:
            print(number)

# Exercise 4

Print a 5x5 square of * characters.


    square = 0
    while square < 5:
        square += 1
        print("*****")

# Exercise 5

Print a NxN square of * characters.


    count = 0
    square = int(input("How big is the square? "))
    while count < square:
        count += 1
        print("*" * square)

# Exercise 6

Given a height and width, input by the user, print a box consisting of * characters as its border. 


    width = int(input("Width? "))
    height = int(input("Height? "))
    empty_space = width - 2
    print("*" * width)
    for i in range(0, height - 2):
        print("*", " " * empty_space, "*", sep="")
    for i in range(height - 2, height - 1):
        print("*" * width)

# Exercise 7

Print a 4 line, centered triangle consisting of * characters. Line 1 will contain 1 * character. 


    lines = 0
    for i in range(0, 4):
        lines += 1
        blank_space = 4 - lines
        increase_char = lines + (lines - 1)
        print(" " * blank_space, "*" * increase_char, sep="")

# Exercise 8

Given a number as the height, print a triangle as in Exercise 7 but with the given height.


    height = int(input("Height? "))
    lines = 0
    for i in range(0, height):
        lines += 1
        blank_space = height - lines
        increase_char = lines + (lines - 1)
        print(" " * blank_space, "*" * increase_char, sep="")

# Exercise 9

Print the multiplication table for numbers from 1 up to 10.


    number1 = 1
    number2 = 0
    while number1 < 11:
        for number in range(0, 10):
            number2 += 1
            result = number1 * number2
            print(f"{number1} X {number2} = {result}")
        number1 += 1
        number2 = 0

# Bonus 1

Given a string, input by the user, print that string with a box around it. The box should stretch to the length of the string.


    text = input("Text? ")
    border = len(text) + 2
    print("*", "*" * border, "*", sep="")
    print("*", text, "*")
    print("*", "*" * border, "*", sep="")

# Bonus 2

Print the first 100 triangle numbers.


    count = 1
    while count <= 100:
        for number in range(0, 100):
            triangle_number = int((count * (count + 1)) / 2)
            count += 1
            print(triangle_number)

# Bonus 3

Given a number, print its factors.


    number = int(input("Pick a number: "))
    divide_by = 0
    factors = []
    while divide_by < number:
        for i in range(0, number):
            divide_by += 1
            divisiable = number % divide_by
            if divisiable == 0:
                factors.append(divide_by)
        print(factors)
